1. Geometry Cleanup Logic
This is the robust cleanup pipeline. It removes "sliver" triangles (degenerate faces) that confuse CSG operations, merges vertices that are virtually identical (welding the mesh), and recalculates normals to ensure the object looks smooth and exports correctly.

// Filters out triangles with zero or near-zero area that cause CSG errors
const removeDegenerateTriangles = (geometry: THREE_ACTUAL.BufferGeometry): THREE_ACTUAL.BufferGeometry => {
    const pos = geometry.attributes.position;
    if (!pos) return geometry;
    const pA = new THREE_ACTUAL.Vector3(); const pB = new THREE_ACTUAL.Vector3(); const pC = new THREE_ACTUAL.Vector3(); 
    const cb = new THREE_ACTUAL.Vector3(); const ab = new THREE_ACTUAL.Vector3();
    const THRESHOLD = 1e-10;
    const newPositions: number[] = [];
    
    // Iterate faces (indexed or non-indexed)
    if (geometry.index) {
        const idx = geometry.index;
        for (let i = 0; i < idx.count; i += 3) {
            const a = idx.getX(i); const b = idx.getX(i + 1); const c = idx.getX(i + 2);
            pA.fromBufferAttribute(pos, a); pB.fromBufferAttribute(pos, b); pC.fromBufferAttribute(pos, c);
            if (isNaN(pA.x) || isNaN(pB.x) || isNaN(pC.x)) { continue; }
            cb.subVectors(pC, pB); ab.subVectors(pA, pB); cb.cross(ab);
            // Only keep if area is sufficient (removes slivers)
            if (cb.lengthSq() > THRESHOLD) { newPositions.push(pA.x, pA.y, pA.z); newPositions.push(pB.x, pB.y, pB.z); newPositions.push(pC.x, pC.y, pC.z); }
        }
    } else {
        for (let i = 0; i < pos.count; i += 3) {
            pA.fromBufferAttribute(pos, i); pB.fromBufferAttribute(pos, i+1); pC.fromBufferAttribute(pos, i+2);
            if (isNaN(pA.x) || isNaN(pB.x) || isNaN(pC.x)) { continue; }
            cb.subVectors(pC, pB); ab.subVectors(pA, pB); cb.cross(ab);
            if (cb.lengthSq() > THRESHOLD) { newPositions.push(pA.x, pA.y, pA.z); newPositions.push(pB.x, pB.y, pB.z); newPositions.push(pC.x, pC.y, pC.z); }
        }
    }
    
    if (newPositions.length > 0) {
        const newGeo = new THREE_ACTUAL.BufferGeometry();
        newGeo.setAttribute('position', new THREE_ACTUAL.Float32BufferAttribute(newPositions, 3));
        return newGeo;
    }
    return geometry;
};

// Robust clean-up: Removes degenerate faces, merges vertices, and recomputes normals
const performGeometryCleanup = (geometry: THREE_ACTUAL.BufferGeometry | null): THREE_ACTUAL.BufferGeometry | null => {
  if (!geometry || !geometry.attributes.position) return null;
  // Clear attributes that get messed up by CSG or merging
  if (geometry.attributes.color) geometry.deleteAttribute('color');
  if (geometry.attributes.uv) geometry.deleteAttribute('uv');
  
  try {
    // 1. Remove slivers
    let clean = removeDegenerateTriangles(geometry);
    // 2. Merge vertices (Tolerance 0.0001mm) - This makes the mesh watertight
    clean = BufferGeometryUtils.mergeVertices(clean, 0.0001);
    // 3. Ensure valid normals for shading/export
    clean.computeVertexNormals();
    return clean;
  } catch (e) { 
    console.warn("Geometry repair failed:", e); 
    return geometry;
  }
};

    const centerX = xOffset + (length - overlap) / 2;
    
    // Tiny random jitter avoids "coplanar faces" (Z-fighting) which crash boolean operations

3. Application of Cuts (Optimized)
This function combines all the cutter blades into one object first (Union), and then subtracts that single object from the snowflake. This is much cleaner and faster than subtracting 10 different blades individually.

const applySlotCuts = async (layerGeo: THREE_ACTUAL.BufferGeometry, layer: LayerConfig, slotLength: number, slotWidth: number, extrusionDepth: number, bevelEnabled: boolean, bevelAmount: number, allLayers: LayerConfig[], evaluator: Evaluator, onProgress?: () => Promise<void>): Promise<THREE_ACTUAL.BufferGeometry> => {
  const slotGeometries = createSlotGeometries(layer, slotLength, slotWidth, extrusionDepth, bevelEnabled, bevelAmount, allLayers);
  if (slotGeometries.length === 0) return layerGeo;

  // 1. Create a single combined Cutting Tool (CSG Union)
  let toolBrush: Brush | null = null;

  for (let i = 0; i < slotGeometries.length; i++) {
    const slotGeo = slotGeometries[i];
    // Align cutter to layer
    slotGeo.rotateX(layer.rotation3D.x * Math.PI / 180);
    slotGeo.rotateY(layer.rotation3D.y * Math.PI / 180);
    
    const brush = new Brush(slotGeo);
    brush.updateMatrixWorld();

    if (!toolBrush) {
        toolBrush = brush;
    } else {
        // Evaluate ADDITION (Union) to combine blades
        const nextTool = evaluator.evaluate(toolBrush, brush, ADDITION);
        if (!slotGeometries.includes(toolBrush.geometry as THREE_ACTUAL.BufferGeometry)) {
            toolBrush.geometry.dispose();
        }
        toolBrush = nextTool;
    }
  }

  if (!toolBrush) return layerGeo;

  // 2. Perform the cut (Subtraction) from the main mesh
  const layerBrush = new Brush(layerGeo);
  layerBrush.updateMatrixWorld();
  toolBrush.updateMatrixWorld();

  const result = evaluator.evaluate(layerBrush, toolBrush, SUBTRACTION);

  // 3. Cleanup Resources
  if (!slotGeometries.includes(toolBrush.geometry as THREE_ACTUAL.BufferGeometry)) {
      toolBrush.geometry.dispose();
  }
  slotGeometries.forEach(g => g.dispose());

  if (onProgress) await onProgress();
  
  return result.geometry;
};
